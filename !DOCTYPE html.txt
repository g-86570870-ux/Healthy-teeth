<script>
let foods = [
  { name:"Milk", good:true, img:"images/milk.png" },
  { name:"Cheese", good:true, img:"images/cheese.png" },
  { name:"Yogurt", good:true, img:"images/yogurt.png" },
  { name:"Apple", good:true, img:"images/apple.png" },
  { name:"Carrot", good:true, img:"images/carrot.png" },
  { name:"Fish", good:true, img:"images/fish.png" },
  { name:"Broccoli", good:true, img:"images/broccoli.png" },
  { name:"Water", good:true, img:"images/water.png" },
  { name:"Candy", good:false, img:"images/candy.png" },
  { name:"Lollipop", good:false, img:"images/lollipop.png" },
  { name:"Soda", good:false, img:"images/soda.png" },
  { name:"Chocolate bar", good:false, img:"images/chocolate.png" },
  { name:"Doughnut", good:false, img:"images/doughnut.png" },
  { name:"Ice cream", good:false, img:"images/icecream.png" },
  { name:"Gummy sweets", good:false, img:"images/gummy.png" },
  { name:"Chips", good:false, img:"images/chips.png" }
];

foods = foods.sort(() => 0.5 - Math.random()).slice(0, 12);

let index = 0;
let score = 0;
let faceLandmarker;
let webcamRunning = false;
let calibrationRoll = 0;

function showInstructions(){
  document.getElementById("welcome-screen").classList.add("hidden");
  document.getElementById("instructions-screen").classList.remove("hidden");
}

async function startWebcam(){
  document.getElementById("instructions-screen").classList.add("hidden");
  document.getElementById("calibration-screen").classList.remove("hidden");

  let video = document.getElementById("video");
  const stream = await navigator.mediaDevices.getUserMedia({ video:true });
  video.srcObject = stream;

  await loadFaceModel();
  setTimeout(calibrate, 2000);
}

async function loadFaceModel(){
  const vision = await FilesetResolver.forVisionTasks(
    "[cdn.jsdelivr.net](https://cdn.jsdelivr.net/npm/@mediapipe/tasks-vision/wasm)"
  );

  faceLandmarker = await FaceLandmarker.createFromOptions(
    vision,
    { baseOptions:{ modelAssetPath:"[cdn.jsdelivr.net](https://cdn.jsdelivr.net/npm/@mediapipe/tasks-vision/face_landmarker.task)" },
      outputFaceBlendshapes:true,
      runningMode:"VIDEO" }
  );
}

function calibrate(){
  detectFrame(true);
}

function startGame(){
  document.getElementById("calibration-screen").classList.add("hidden");
  document.getElementById("game-screen").classList.remove("hidden");
  nextQuestion();
  webcamRunning = true;
  detectFrame(false);
}

function nextQuestion(){
  if(index >= foods.length){
    showFinal();
    return;
  }

  let item = foods[index];
  document.getElementById("food-img").src = item.img;
  document.getElementById("feedback").textContent = "";
  updateProgress();
}

function updateProgress(){
  let percent = (index / foods.length) * 100;
  document.getElementById("progress-bar").style.width = percent + "%";
}

function detectFrame(isCalibrating){
  let video = document.getElementById("video");

  function frame(){
    if(!faceLandmarker) return;
    
    const face = faceLandmarker.detectForVideo(video, performance.now());
    if(face?.faceGeometry?.poseTransformationMatrix){
      let roll = face.faceGeometry.poseTransformationMatrix.roll * (180/Math.PI);

      if(isCalibrating){
        calibrationRoll = roll;
        startGame();
        return;
      }

      let relative = roll - calibrationRoll;

      if(relative < -12) evaluate(true);  
      if(relative > 12) evaluate(false); 
    }
    requestAnimationFrame(frame);
  }

  frame();
}

function evaluate(playerThinksGood){
  let correct = foods[index].good === playerThinksGood;

  if(correct){
    score++;
    showFeedback("Correct! ⭐", "stars");
  } else {
    showFeedback("Oops!", "");
  }

  index++;
  setTimeout(nextQuestion, 1000);
}

function showFeedback(text, effectClass){
  let el = document.getElementById("feedback");
  el.textContent = text;
  el.className = effectClass;
  setTimeout(()=> el.className="", 500);
}

function showFinal(){
  document.getElementById("game-screen").classList.add("hidden");
  document.getElementById("final-screen").classList.remove("hidden");
  document.getElementById("final-score").textContent =
    `You scored ${score} out of ${foods.length}!`;
}
